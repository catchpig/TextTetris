> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a TextTetris Android game with Compose + Canvas, featuring gesture controls, Room database for game records, and vibrant color text-based tetromino rendering.

**Architecture:** MVVM + DDD with three layers (UI/Domain/Data). UI layer handles Compose screens and Canvas drawing. Domain layer contains game logic (board, pieces, collision, scoring). Data layer manages Room database and game record persistence.

**Tech Stack:** Kotlin, Jetpack Compose, Compose Canvas API, Room, Coroutines + Flow, Hilt (optional DI)

```kotlin
    val blocks: List<Position>,
    val position: Position,
    val rotation: Int = 0
) {
    companion object {
        private val SHAPES = mapOf(
            TetrominoType.I to listOf(
                listOf(Position(0,1), Position(1,1), Position(2,1), Position(3,1)),
                listOf(Position(2,0), Position(2,1), Position(2,2), Position(2,3)),
                listOf(Position(0,2), Position(1,2), Position(2,2), Position(3,2)),
                listOf(Position(1,0), Position(1,1), Position(1,2), Position(1,3))
            ),
            TetrominoType.O to listOf(
                listOf(Position(1,0), Position(2,0), Position(1,1), Position(2,1))
            ),
            TetrominoType.T to listOf(
                listOf(Position(1,0), Position(0,1), Position(1,1), Position(2,1)),
                listOf(Position(1,0), Position(1,1), Position(2,1), Position(1,2)),
                listOf(Position(0,1), Position(1,1), Position(2,1), Position(1,2)),
                listOf(Position(1,0), Position(0,1), Position(1,1), Position(1,2))
            ),
            TetrominoType.S to listOf(
                listOf(Position(1,0), Position(2,0), Position(0,1), Position(1,1)),
                listOf(Position(1,0), Position(1,1), Position(2,1), Position(2,2)),
                listOf(Position(1,1), Position(2,1), Position(0,2), Position(1,2)),
                listOf(Position(0,0), Position(0,1), Position(1,1), Position(1,2))
            ),
            TetrominoType.Z to listOf(
                listOf(Position(0,0), Position(1,0), Position(1,1), Position(2,1)),
                listOf(Position(2,0), Position(1,1), Position(2,1), Position(1,2)),
                listOf(Position(0,1), Position(1,1), Position(1,2), Position(2,2)),
                listOf(Position(1,0), Position(0,1), Position(1,1), Position(0,2))
            ),
            TetrominoType.J to listOf(
                listOf(Position(0,0), Position(0,1), Position(1,1), Position(2,1)),
                listOf(Position(1,0), Position(2,0), Position(1,1), Position(1,2)),
                listOf(Position(0,1), Position(1,1), Position(2,1), Position(2,2)),
                listOf(Position(1,0), Position(1,1), Position(0,2), Position(1,2))
            ),
            TetrominoType.L to listOf(
                listOf(Position(2,0), Position(0,1), Position(1,1), Position(2,1)),
                listOf(Position(1,0), Position(1,1), Position(1,2), Position(2,2)),
                listOf(Position(0,1), Position(1,1), Position(2,1), Position(0,2)),
                listOf(Position(0,0), Position(1,0), Position(1,1), Position(1,2))
            )
        )

        fun create(type: TetrominoType, position: Position = Position(3, 0)): Tetromino {
            val blocks = SHAPES[type]?.first() ?: emptyList()
            return Tetromino(type, blocks, position, 0)
        }
    }

    fun moveTo(position: Position) = copy(position = position)
    fun rotate(): Tetromino {
        val shapeIndex = rotation % 4
        val nextRotation = (rotation + 1) % 4
        val blocks = SHAPES[type]?.get(nextRotation) ?: emptyList()
        return copy(blocks = blocks, rotation = nextRotation)
    }

    fun absoluteBlocks(): List<Position> = blocks.map { it + position }
}
```

```kotlin
package com.texttetris.domain.model

data class GameBoard(
    val width: Int = 10,
    val height: Int = 20,
    val cells: List<MutableList<Cell>> = List(height) { MutableList(width) { Cell() } }
) {
    fun isValidPosition(pos: Position): Boolean =
        pos.x in 0 until width && pos.y in 0 until height

    fun isEmptyPosition(pos: Position): Boolean =
        isValidPosition(pos) && cells[pos.y][pos.x].color == null

    fun setCell(pos: Position, color: TetrominoColor): GameBoard {
        if (!isValidPosition(pos)) return this
        val newCells = cells.map { it.toMutableList() }
        newCells[pos.y][pos.x] = Cell(color)
        return copy(cells = newCells)
    }

    fun clearFullLines(): Pair<GameBoard, Int> {
        val fullLines = cells.indices.filter { row -> cells[row].all { it.color != null } }
        if (fullLines.isEmpty()) return Pair(this, 0)

        val newCells = cells
            .filterIndexed { index, _ -> index !in fullLines }
            .toMutableList()
        repeat(fullLines.size) { newCells.add(0, MutableList(width) { Cell() }) }

        return Pair(copy(cells = newCells), fullLines.size)
    }

    companion object {
        fun create() = GameBoard()
    }
}
```

```kotlin
package com.texttetris.domain.model

object Score {
    private val SCORE_TABLE = mapOf(
        1 to 100,
        2 to 300,
        3 to 500,
        4 to 800
    )

    fun calculate(linesCleared: Int, level: Int): Int {
        val baseScore = SCORE_TABLE[linesCleared] ?: (linesCleared * 100)
        return baseScore * level
    }

    fun levelUp(linesCleared: Int, currentLevel: Int): Int {
        val linesPerLevel = 10
        return currentLevel + (linesCleared / linesPerLevel)
    }
}
```

```kotlin
package com.texttetris.domain.usecase

import com.texttetris.domain.model.GameBoard
import com.texttetris.domain.model.Position
import com.texttetris.domain.model.Tetromino

class MoveTetrominoUseCase {
    fun moveLeft(tetromino: Tetromino, board: GameBoard): Tetromino {
        val newPos = tetromino.position + Position(-1, 0)
        val newTetromino = tetromino.moveTo(newPos)
        return if (canPlace(newTetromino, board)) newTetromino else tetromino
    }

    fun moveRight(tetromino: Tetromino, board: GameBoard): Tetromino {
        val newPos = tetromino.position + Position(1, 0)
        val newTetromino = tetromino.moveTo(newPos)
        return if (canPlace(newTetromino, board)) newTetromino else tetromino
    }

    fun moveDown(tetromino: Tetromino, board: GameBoard): Tetromino {
        val newPos = tetromino.position + Position(0, 1)
        val newTetromino = tetromino.moveTo(newPos)
        return if (canPlace(newTetromino, board)) newTetromino else tetromino
    }

    fun canPlace(tetromino: Tetromino, board: GameBoard): Boolean {
        return tetromino.absoluteBlocks().all { board.isEmptyPosition(it) }
    }

    fun canMoveDown(tetromino: Tetromino, board: GameBoard): Boolean {
        val newPos = tetromino.position + Position(0, 1)
        val newTetromino = tetromino.moveTo(newPos)
        return canPlace(newTetromino, board)
    }
}
```

```kotlin
package com.texttetris.domain.usecase

import com.texttetris.domain.model.GameBoard
import com.texttetris.domain.model.Tetromino

class RotateTetrominoUseCase {
    private val moveUseCase = MoveTetrominoUseCase()

    fun rotate(tetromino: Tetromino, board: GameBoard): Tetromino {
        val rotated = tetromino.rotate()
        return if (moveUseCase.canPlace(rotated, board)) {
            rotated
        } else {
            // Try wall kicks
            val kicks = listOf(
                tetromino.position + com.texttetris.domain.model.Position(-1, 0),
                tetromino.position + com.texttetris.domain.model.Position(1, 0),
                tetromino.position + com.texttetris.domain.model.Position(0, -1)
            )
            kicks.fold(rotated) { acc, pos ->
                if (moveUseCase.canPlace(acc.copy(position = pos), board)) {
                    acc.copy(position = pos)
                } else acc
            }
        }
    }
}
```

```kotlin
package com.texttetris.data.local

import android.content.Context
import androidx.room.Database
import androidx.room.Room
import androidx.room.RoomDatabase

@Database(entities = [GameRecordEntity::class], version = 1)
abstract class GameDatabase : RoomDatabase() {
    abstract fun gameRecordDao(): GameRecordDao

    companion object {
        @Volatile
        private var INSTANCE: GameDatabase? = null

        fun getInstance(context: Context): GameDatabase {
            return INSTANCE ?: synchronized(this) {
                val instance = Room.databaseBuilder(
                    context.applicationContext,
                    GameDatabase::class.java,
                    "texttetris_database"
                ).build()
                INSTANCE = instance
                instance
            }
        }
    }
}
```

```kotlin
package com.texttetris.ui.viewmodel

import androidx.lifecycle.ViewModel
import androidx.lifecycle.ViewModelProvider
import androidx.lifecycle.viewModelScope
import com.texttetris.data.local.GameRecordEntity
import com.texttetris.domain.model.GameBoard
import com.texttetris.domain.model.GameState
import com.texttetris.domain.model.Tetromino
import com.texttetris.domain.repository.GameRepository
import com.texttetris.domain.usecase.*
import kotlinx.coroutines.Job
import kotlinx.coroutines.delay
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.launch

class GameViewModel(
    private val repository: GameRepository
) : ViewModel() {

    private val moveUseCase = MoveTetrominoUseCase()
    private val rotateUseCase = RotateTetrominoUseCase()
    private val spawnUseCase = SpawnTetrominoUseCase()
    private val lockUseCase = LockTetrominoUseCase()
    private val clearUseCase = ClearLinesUseCase()
    private val scoreUseCase = CalculateScoreUseCase()
    private val gameOverUseCase = CheckGameOverUseCase()

    private val _gameState = MutableStateFlow(GameState.IDLE)
    val gameState: StateFlow<GameState> = _gameState.asStateFlow()

    private val _board = MutableStateFlow(GameBoard.create())
    val board: StateFlow<GameBoard> = _board.asStateFlow()

    private val _currentTetromino = MutableStateFlow<Tetromino?>(null)
    val currentTetromino: StateFlow<Tetromino?> = _currentTetromino.asStateFlow()

    private val _nextTetromino = MutableStateFlow<Tetromino?>(null)
    val nextTetromino: StateFlow<Tetromino?> = _nextTetromino.asStateFlow()

    private val _score = MutableStateFlow(0)
    val score: StateFlow<Int> = _score.asStateFlow()

    private val _level = MutableStateFlow(1)
    val level: StateFlow<Int> = _level.asStateFlow()

    private val _linesCleared = MutableStateFlow(0)
    val linesCleared: StateFlow<Int> = _linesCleared.asStateFlow()

    private val _highScore = MutableStateFlow(0)
    val highScore: StateFlow<Int> = _highScore.asStateFlow()

    private var gameLoopJob: Job? = null
    private var gameStartTime: Long = 0

    init {
        viewModelScope.launch {
            _highScore.value = repository.getHighScore()
        }
    }

    fun startGame() {
        _board.value = GameBoard.create()
        _score.value = 0
        _level.value = 1
        _linesCleared.value = 0
        _currentTetromino.value = spawnUseCase.spawn()
        _nextTetromino.value = spawnUseCase.spawn()
        _gameState.value = GameState.PLAYING
        gameStartTime = System.currentTimeMillis()
        startGameLoop()
    }

    fun pauseGame() {
        _gameState.value = GameState.PAUSED
        gameLoopJob?.cancel()
    }

    fun resumeGame() {
        _gameState.value = GameState.PLAYING
        startGameLoop()
    }

    fun moveLeft() {
        _currentTetromino.value?.let { tetromino ->
            _currentTetromino.value = moveUseCase.moveLeft(tetromino, _board.value)
        }
    }

    fun moveRight() {
        _currentTetromino.value?.let { tetromino ->
            _currentTetromino.value = moveUseCase.moveRight(tetromino, _board.value)
        }
    }

    fun rotate() {
        _currentTetromino.value?.let { tetromino ->
            _currentTetromino.value = rotateUseCase.rotate(tetromino, _board.value)
        }
    }

    fun drop() {
        while (moveUseCase.canMoveDown(_currentTetromino.value!!, _board.value)) {
            _currentTetromino.value = _currentTetromino.value?.let {
                moveUseCase.moveDown(it, _board.value)
            }
        }
        lockCurrentPiece()
    }

    private fun startGameLoop() {
        gameLoopJob?.cancel()
        gameLoopJob = viewModelScope.launch {
            while (true) {
                delay(getDropInterval().toLong())
                tick()
            }
        }
    }

    private fun tick() {
        val tetromino = _currentTetromino.value ?: return

        if (moveUseCase.canMoveDown(tetromino, _board.value)) {
            _currentTetromino.value = moveUseCase.moveDown(tetromino, _board.value)
        } else {
            lockCurrentPiece()
        }
    }

    private fun lockCurrentPiece() {
        val tetromino = _currentTetromino.value ?: return

        if (gameOverUseCase.isGameOver(tetromino, _board.value)) {
            endGame()
            return
        }

        _board.value = lockUseCase.lock(tetromino, _board.value)

        val (newBoard, lines) = clearUseCase.clear(_board.value)
        _board.value = newBoard

        if (lines > 0) {
            _linesCleared.value += lines
            _score.value = scoreUseCase.addScore(_score.value, lines, _level.value)
            _level.value = scoreUseCase.levelUp(_linesCleared.value, _level.value)
        }

        _currentTetromino.value = _nextTetromino.value
        _nextTetromino.value = spawnUseCase.spawn()

        if (gameOverUseCase.isGameOver(_currentTetromino.value!!, _board.value)) {
            endGame()
        }
    }

    private fun endGame() {
        _gameState.value = GameState.GAME_OVER
        gameLoopJob?.cancel()

        viewModelScope.launch {
            val record = GameRecordEntity(
                score = _score.value,
                level = _level.value,
                linesCleared = _linesCleared.value,
                duration = System.currentTimeMillis() - gameStartTime,
                playedAt = System.currentTimeMillis()
            )
            repository.saveGameRecord(record)
            _highScore.value = repository.getHighScore()
        }
    }

    private fun getDropInterval(): Int {
        return when {
            _level.value >= 15 -> 100
            _level.value >= 10 -> 200
            _level.value >= 5 -> 400
            else -> 800
        }
    }
}
```

```kotlin
@Composable
fun GameBoardCanvas(
    board: GameBoard,
    currentTetromino: Tetromino?,
    modifier: Modifier = Modifier
) {
    val density = LocalDensity.current
    val cellSize = with(density) { 20.dp.toPx() }

    Box(
        modifier = modifier
            .size(
                width = (board.width * cellSize).dp.value.dp,
                height = (board.height * cellSize).dp.value.dp
            )
            .background(Surface)
            .border(1.dp, NeonCyan),
        contentAlignment = Alignment.Center
    ) {
        Canvas(modifier = Modifier.fillMaxSize()) {
            // Draw board cells
            board.cells.forEachIndexed { y, row ->
                row.forEachIndexed { x, cell ->
                    val color = cell.color?.toComposeColor() ?: Color.Transparent
                    drawCell(
                        x = x,
                        y = y,
                        color = color,
                        cellSize = cellSize,
                        filled = cell.color != null
                    )
                }
            }

            // Draw current tetromino
            currentTetromino?.absoluteBlocks()?.forEach { pos ->
                drawCell(
                    x = pos.x,
                    y = pos.y,
                    color = currentTetromino.type.color.toComposeColor(),
                    cellSize = cellSize,
                    filled = true
                )
            }
        }
    }
}

private fun DrawScope.drawCell(
    x: Int,
    y: Int,
    color: Color,
    cellSize: Float,
    filled: Boolean
) {
    val padding = 1f
    val topLeft = Offset(x * cellSize + padding, y * cellSize + padding)
    val cellInnerSize = cellSize - padding * 2

    if (filled) {
        // Filled cell with neon color
        drawRect(
            color = color,
            topLeft = topLeft,
            size = androidx.compose.ui.geometry.Size(cellInnerSize, cellInnerSize)
        )
        // Glow effect
        drawRect(
            color = color.copy(alpha = 0.3f),
            topLeft = Offset(topLeft.x - 2, topLeft.y - 2),
            size = androidx.compose.ui.geometry.Size(cellInnerSize + 4, cellInnerSize + 4)
        )
    } else {
        // Empty cell - hollow square
        drawRect(
            color = Color(0xFF333333),
            topLeft = topLeft,
            size = androidx.compose.ui.geometry.Size(cellInnerSize, cellInnerSize),
            style = Stroke(width = 1f)
        )
    }
}

    TetrominoColor.PURPLE -> NeonPurple
    TetrominoColor.GREEN -> NeonGreen
    TetrominoColor.RED -> NeonRed
    TetrominoColor.BLUE -> NeonBlue
    TetrominoColor.ORANGE -> NeonOrange
```

```kotlin
package com.texttetris.ui.components

import androidx.compose.foundation.gestures.detectDragGestures
import androidx.compose.foundation.gestures.detectTapGestures
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.geometry.Offset
import androidx.compose.ui.input.pointer.pointerInput
import kotlin.math.abs

@Composable
fun Modifier.gestureDetector(
    onSwipeLeft: () -> Unit,
    onSwipeRight: () -> Unit,
    onSwipeUp: () -> Unit,
    onSwipeDown: () -> Unit,
    onDoubleTap: () -> Unit
): Modifier {
    var startOffset by remember { mutableStateOf(Offset.Zero) }

    return this.pointerInput(Unit) {
        detectTapGestures(
            onDoubleTap = { onDoubleTap() }
        )
    }.pointerInput(Unit) {
        detectDragGestures(
            onDragStart = { offset -> startOffset = offset },
            onDragEnd = {},
            onDrag = { change, _ ->
                val delta = change.position - startOffset
                val threshold = 50f

                when {
                    abs(delta.x) > abs(delta.y) && abs(delta.x) > threshold -> {
                        if (delta.x > 0) onSwipeRight() else onSwipeLeft()
                        startOffset = change.position
                    }
                    abs(delta.y) > threshold -> {
                        if (delta.y > 0) onSwipeDown() else onSwipeUp()
                        startOffset = change.position
                    }
                }
            }
        )
    }
}
```

```kotlin
        moveSound = soundPool.load(context, R.raw.move, 1)
        rotateSound = soundPool.load(context, R.raw.rotate, 1)
        clearSound = soundPool.load(context, R.raw.clear, 1)
        gameOverSound = soundPool.load(context, R.raw.gameover, 1)

    fun playMove() { soundPool.play(moveSound, 0.5f, 0.5f, 1, 0, 1f) }
    fun playRotate() { soundPool.play(rotateSound, 0.3f, 0.3f, 1, 0, 1f) }
    fun playClear() { soundPool.play(clearSound, 0.7f, 0.7f, 1, 0, 1f) }
    fun playGameOver() { soundPool.play(gameOverSound, 1f, 1f, 1, 0, 0.8f) }

    fun release() { soundPool.release() }
}
```
